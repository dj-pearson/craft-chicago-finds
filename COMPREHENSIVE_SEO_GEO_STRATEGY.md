**Product Sitemap (`/sitemap-products.xml`):**
```typescript
// /api/sitemap-products.ts
import { createClient } from '@supabase/supabase-js';

export async function GET() {
  const supabase = createClient(
    process.env.SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_KEY!
  );

  // Fetch all live listings
  const { data: listings, error } = await supabase
    .from('listings')
    .select('slug, updated_at, city_id, category')
    .eq('status', 'live')
    .order('updated_at', { ascending: false })
    .limit(50000); // Google limit

  if (error) throw error;

  const baseUrl = 'https://craftchicagofinds.com';
  
  const urlset = listings.map(listing => {
    // Determine priority based on factors
    const priority = calculatePriority(listing);
    const changefreq = determineChangeFreq(listing);
    
    return `
  <url>
    <loc>${baseUrl}/chicago/product/${listing.category}/${listing.slug}</loc>
    <lastmod>${new Date(listing.updated_at).toISOString()}</lastmod>
    <changefreq>${changefreq}</changefreq>
    <priority>${priority}</priority>
    <image:image>
      <image:loc>${baseUrl}/images/products/${listing.slug}-1.jpg</image:loc>
      <image:caption>Handmade ${listing.category}</image:caption>
    </image:image>
  </url>`;
  }).join('');

  const sitemap = `<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9"
        xmlns:image="http://www.google.com/schemas/sitemap-image/1.1">
${urlset}
</urlset>`;

  return new Response(sitemap, {
    headers: {
      'Content-Type': 'application/xml',
      'Cache-Control': 'public, s-maxage=3600, stale-while-revalidate'
    }
  });
}

function calculatePriority(listing: any): number {
  // High priority for:
  // - Recently updated
  // - High-rated makers
  // - Popular categories
  let priority = 0.5; // Base priority
  
  const daysSinceUpdate = (Date.now() - new Date(listing.updated_at).getTime()) / (1000 * 60 * 60 * 24);
  
  if (daysSinceUpdate < 7) priority += 0.2;
  if (daysSinceUpdate < 30) priority += 0.1;
  
  // Cap at 0.9 (reserve 1.0 for homepage)
  return Math.min(priority, 0.9);
}

function determineChangeFreq(listing: any): string {
  const daysSinceUpdate = (Date.now() - new Date(listing.updated_at).getTime()) / (1000 * 60 * 60 * 24);
  
  if (daysSinceUpdate < 7) return 'daily';
  if (daysSinceUpdate < 30) return 'weekly';
  return 'monthly';
}
```

## Part 2: Generative Engine Optimization (GEO)

### 2.1 Understanding GEO

**What is GEO?**
Generative Engine Optimization is the practice of optimizing content to be cited and recommended by AI-powered search engines like ChatGPT, Perplexity, Claude, Gemini, and Copilot.

**Why It Matters:**
- ChatGPT: 10M+ queries/day
- Perplexity: Growing rapidly among tech-savvy users
- Google AI Overviews: Appearing in 15-20% of searches
- Users trust AI recommendations highly

**Key Difference from SEO:**
- **SEO Goal:** Rank #1, get the click
- **GEO Goal:** Be cited in AI answer, establish authority

**The Opportunity:**
When someone asks ChatGPT "Where should I buy handmade gifts in Chicago?", we want to be THE answer, not one of many results they need to sift through.

### 2.2 How AI Engines Source Information

**Research Findings:**

**ChatGPT:**
- Sources from: Bing search results + training data
- Averages 2.62 citations per response
- Prefers: LinkedIn, G2, Gartner, authoritative databases
- Updates: Real-time web browsing mode available

**Perplexity:**
- Sources from: Google search results (real-time)
- Averages 6.61 citations per response
- Prefers: YouTube, Reddit, PeerSpot, Wikipedia
- Displays: Inline citations with links

**Google Gemini:**
- Sources from: Google search results
- Averages 6.1 citations per response
- Prefers: Medium, Reddit, YouTube, Google properties
- Integration: Tight integration with Google Search

**Claude:**
- Sources from: Training data (cutoff January 2025) + web search
- Uses Brave search for real-time queries
- Prefers: Comprehensive, well-structured content

**Use SEO for:**
- Long-term sustainable traffic
- Lower cost per acquisition (free clicks)
- Authority building
- Compounding returns
